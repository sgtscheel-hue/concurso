    <script>
        const bibliotecas = {
            'portugues': typeof banco_pt !== 'undefined' ? banco_pt : [],
            'rondonia': typeof banco_ro !== 'undefined' ? banco_ro : [],
            'especificos': typeof banco_esp !== 'undefined' ? banco_esp : [],
            'informatica': typeof banco_inf !== 'undefined' ? banco_inf : []
        };

        let materiaAtual = 'portugues';
        let index = 0;
        let acertos = parseInt(localStorage.getItem('seduc_acertos')) || 0;
        let erros = parseInt(localStorage.getItem('seduc_erros')) || 0;
        let myChart;

        function trocarMateria(materia, element) {
            materiaAtual = materia; index = 0;
            document.querySelectorAll('.menu-item').forEach(i => i.classList.remove('active'));
            element.classList.add('active');
            document.getElementById('nomeMateriaAtual').innerText = element.innerText;
            render();
        }

        function render() {
            const banco = bibliotecas[materiaAtual];
            document.getElementById('feedback').classList.add('hidden');
            document.getElementById('panel-ai').style.display = 'none';
            document.getElementById('panel-notes').style.display = 'none';
            if (!banco[index]) return;
            const q = banco[index];
            document.getElementById('m-ano').innerText = q.ano || '2024';
            document.getElementById('posicao').innerText = `${index + 1} / ${banco.length}`;
            document.getElementById('enunciado').innerText = q.enunciado;
            document.getElementById('alternativas').innerHTML = ['a','b','c','d','e'].map(l => {
                if(!q[l]) return '';
                return `<label class="block"><input type="radio" name="quiz" value="${l}" class="hidden"><div class="alternativa-card"><span class="letra-circulo uppercase">${l}</span><div class="text-gray-700 text-[13px]">${q[l]}</div></div></label>`;
            }).join('');
            document.getElementById('user-notes').value = localStorage.getItem(`note_${materiaAtual}_${index}`) || "";
            updateDash();
        }

        function verificar() {
            const banco = bibliotecas[materiaAtual];
            const sel = document.querySelector('input[name="quiz"]:checked');
            if(!sel) return;
            const correta = banco[index].correta.toLowerCase().trim();
            const f = document.getElementById('feedback');
            f.classList.remove('hidden');
            document.getElementById('panel-ai').style.display = 'block';

            if(sel.value === correta) {
                acertos++; f.className = "mt-4 p-3 rounded-lg border-l-4 bg-emerald-50 border-emerald-500 text-emerald-800 text-xs";
                f.innerHTML = `<strong>✓ Correto!</strong>`;
            } else {
                erros++; f.className = "mt-4 p-3 rounded-lg border-l-4 bg-red-50 border-red-500 text-red-800 text-xs";
                f.innerHTML = `<strong>✕ Errado. Gabarito: ${correta.toUpperCase()}</strong>`;
            }
            localStorage.setItem('seduc_acertos', acertos);
            localStorage.setItem('seduc_erros', erros);
            document.getElementById('ai-resumo').innerText = banco[index].comentario || "Conceito fundamental abordado.";
            updateDash();
        }

        function updateDash() {
            document.getElementById('dashAcertos').innerText = acertos;
            document.getElementById('dashErros').innerText = erros;
            updateChart();
        }

        function updateChart() {
            const total = acertos + erros;
            const ctx = document.getElementById('performanceChart').getContext('2d');
            const data = {
                labels: ['Acertos', 'Erros'],
                datasets: [{
                    data: total === 0 ? [1, 0] : [acertos, erros],
                    backgroundColor: total === 0 ? ['#E5E7EB', '#E5E7EB'] : ['#10B981', '#EF4444'],
                    borderWidth: 0
                }]
            };

            if (myChart) {
                myChart.data = data;
                myChart.update();
            } else {
                myChart = new Chart(ctx, {
                    type: 'doughnut',
                    data: data,
                    options: {
                        cutout: '70%',
                        plugins: {
                            legend: { display: false },
                            tooltip: { enabled: total > 0 }
                        }
                    }
                });
            }
        }

        function resetar() { 
            if(confirm("Zerar desempenho?")) { 
                acertos = 0; erros = 0; 
                localStorage.setItem('seduc_acertos', 0);
                localStorage.setItem('seduc_erros', 0);
                updateDash(); 
            } 
        }

        function togglePanel(id) {
            const p = document.getElementById(id);
            p.style.display = p.style.display === 'block' ? 'none' : 'block';
        }

        document.getElementById('user-notes').addEventListener('input', (e) => {
            localStorage.setItem(`note_${materiaAtual}_${index}`, e.target.value);
        });

        function proxima() { if(index < bibliotecas[materiaAtual].length - 1) { index++; render(); } }
        function anterior() { if(index > 0) { index--; render(); } }

        window.onload = () => render();
    </script>
